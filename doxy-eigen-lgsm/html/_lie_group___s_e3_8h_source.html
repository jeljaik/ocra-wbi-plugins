204 // log
205 template<class Derived>
206 typename LieGroupBase<Array<typename internal::traits<Derived>::Scalar, 7, 1 >, Derived>::Algebra
207  LieGroupBase<Array<typename internal::traits<Derived>::Scalar, 7, 1 >, Derived>::log(const Scalar precision) const
208 {
209  typename SO3Element::Algebra ang = this->getSO3Element().log(precision);
210 
211  const Scalar n2 = ang.squaredNorm(); // ||wtidle||^2
212 
213  if(n2 < precision)
214  return Algebra(ang, this->getR3Element());
215  else{
216  const Scalar n = ::sqrt(n2);
217  const Scalar sn = ::sin(n);
218 
219  Scalar val = (Scalar(2.0) * sn - n * (Scalar(1.0) + ::cos(n))) / (Scalar(2.0) *n2 * sn);
220 
221  Vector3 lin = -0.5*ang.cross(this->getR3Element());
222  lin += (Scalar(1.0) - val * n2 ) * this->getR3Element();
223  lin += val * ang.dot(this->getR3Element()) * ang;
224 
225  return Algebra(ang, lin);
226  }
227 }

229 // adjoint
230 //
231 // Ad_H = [ R 0 ]
232 // [ ptilde*R R ]
233 //
234 template<class Derived>
235 typename LieGroupBase<Array<typename internal::traits<Derived>::Scalar, 7, 1 >, Derived>::AdjointMatrix
236  LieGroupBase<Array<typename internal::traits<Derived>::Scalar, 7, 1 >, Derived>::adjoint() const
237 {
238  AdjointMatrix res;
239 
240  res.template block<3,3>(0,0) = this->getSO3Element().adjoint();
241  res.template block<3,3>(3,3) = res.template block<3,3>(0,0);
242 
243 
244  res.template block<3,3>(3,0) = (Matrix<Scalar,3,3>() <<
245  0, -this->getR3Element()[2], this->getR3Element()[1], // 0 -z y
246  this->getR3Element()[2], 0, -this->getR3Element()[0], // z 0 -x
247  -this->getR3Element()[1], this->getR3Element()[0], 0) //-y x 0
248  .finished()*res.template block<3,3>(0,0);
249 
250  (res.template block<3,3>(0,3)).setZero();
251 
252  return res;
253 }

255 //
256 // Ad_H * T = [ R 0 ] * [ w ] = [ R*w ]
257 // [ ptilde*R R ] [ v ] [ (p^R*w)+R*v ]
258 //
259 template<class Derived>
260 template<class AlgebraDerived>
261 inline typename LieGroupBase<Array<typename internal::traits<Derived>::Scalar, 7, 1 >, Derived>::Algebra
262  LieGroupBase<Array<typename internal::traits<Derived>::Scalar, 7, 1 >, Derived>::adjoint(const LieAlgebraBase<Matrix<Scalar, 6, 1>, AlgebraDerived>& a) const
263 {
264  Vector3 Rw = this->getSO3Element() * a.getso3Element(); // R*w
265 
266  typename SO3Element::Algebra ang(Rw);
267  Vector3 lin = this->getR3Element().cross(Rw) + this->getSO3Element() * a.getR3Element(); // (p^R*w)+R*v
268 
269  return Algebra(ang, lin);
270 }

185 //log
186 template<class Derived>
187 typename LieGroupBase<Quaternion<typename internal::traits<Derived>::Scalar>, Derived>::Algebra
188  LieGroupBase<Quaternion<typename internal::traits<Derived>::Scalar>, Derived>::log(const Scalar precision) const
189 {
190  const Scalar n2 = this->get().vec().squaredNorm();
191  const Scalar n = ::sqrt(n2);
192 
193  if (n < precision)
194  return Algebra((2 / this->get().w()) * this->get().vec());
195  else
196  return Algebra(::atan2(2 * n * this->get().w(), this->get().w() * this->get().w() - n2) / n * this->get().vec());
197 }
